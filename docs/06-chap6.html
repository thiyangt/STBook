  1. stats::decompose(ts_data, type = c(“additive”, “multiplicative”)) Decompose a time series into seasonal, trend and irregular components using moving average.

library(expsmooth) # use a dataset cangas here
library(ggplot2)
ts_data = cangas
ts_data |> ggAcf()
autoplot(ts_data) + ylab("Canadaian Gas Production")
fit <- decompose(ts_data, type='additive')
autoplot(fit)
